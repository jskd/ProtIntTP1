MAKEFILE RULES :
----------------------------------------------------
 - make : Compile dans bin/
 - make runservtcp : Execute le serveur TCP
 - make runclitcp : Execute le client TCP
 - make runservudp : Execute le serveur UDP
 - make runcliudp : Execute le client UDP
 - make tar : Archive le projet
 - make clean : Supprime les .class
